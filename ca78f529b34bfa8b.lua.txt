    local function setProgress(val, text)
        StatusLabel.Text = text
        TweenService:Create(Bar, TweenInfo.new(0.5), {Size = UDim2.new(val, 0, 1, 0)}):Play()
        task.wait(0.6)
    end

    setProgress(0.2, "Checking Local Database...")
    if isfile and isfile(XAEL_CONFIG.PERM_BAN_FILE) then
        UI:Destroy()
        lp:Kick("XAEL5 â€¢ Local HWID Ban Detected.")
        return
    end

    setProgress(0.4, "Verifying Game...")
    local gameInfo = {}
    local successName = pcall(function() gameInfo = MarketplaceService:GetProductInfo(placeId) end)
    local gameName = (successName and gameInfo.Name) or ""

    if not string.find(string.lower(gameName), string.lower(XAEL_CONFIG.REQUIRED_GAME)) then
        UI:Destroy()
        lp:Kick("XAEL5 â€¢ Unsupported Game: " .. gameName)
        return
    end

    setProgress(0.6, "Checking Blacklist...")
    local successBL, blacklistContent = pcall(game.HttpGet, game, XAEL_CONFIG.BANLIST)
    if successBL then
        if string.find(blacklistContent, tostring(lp.UserId)) or string.find(blacklistContent, hwid) then
            UI:Destroy()
            lp:Kick("XAEL5 â€¢ Globally Blacklisted.")
            return
        end
    end

    setProgress(0.8, "Connecting to Sentinel Unit...")
    notifySentinelBot("SESSION_START", 0x00FF9D)

    setProgress(1.0, "Fetching Script Core...")
    local successScript, scriptData = pcall(game.HttpGet, game, XAEL_CONFIG.MAIN_SCRIPT)
    
    if successScript and #scriptData > 100 then
        for name, func in pairs(originals) do
            getgenv()[name] = func
        end

        local main_load, load_err = loadstring(scriptData, "XAEL5_CORE")
        if main_load then
            task.spawn(function()
                pcall(main_load)
            end)
            StatusLabel.Text = "Loaded Successfully!"
            task.wait(1)
            UI:Destroy()
        else
            StatusLabel.Text = "Load Error!"
            task.wait(2)
            UI:Destroy()
        end
    else
        UI:Destroy()
        lp:Kick("XAEL5 â€¢ Connection Failed.")
    end
end
